nums = [1, 1, 1, 0, 1, 1]


def findmaxconsec(nums):    
    count = 0
    count_max = 0
    for item in nums:
        if item == 1:
            count += 1
        else:
            if count > count_max:
                count_max = count
            count = 0
    if count > count_max:
        count_max = count
    return count_max


print(findmaxconsec(nums))
